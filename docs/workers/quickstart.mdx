```ts registry.ts
import { setup } from "rivetkit";
import { worker } from "rivetkit/worker";

const counter = worker({
	onAuth: async () => {
		// Allow public access
	},
	state: {
		count: 0,
	},
	actions: {
		foo: (c) => c.state.count++,
	},
});

const myWorkflow = workflow({
	run: () => {}
});

const myRealtime = realtime({
	run: () => {}
});

const registry = setup({
	registry: { counter, myWorkflow, myRealtime },
	maxConnParamLength: 123
});

export type Registry = typeof registry;
```

```ts server.ts
// With router
import { registry } from "./registry";

const app = new Hono();
app.route("/registry", c => registry.handler(c.req.raw));
serve(app);

// Without router
import { serve } from "@rivetkit/node";

serve(registry);
```

```ts client.ts
import { createClient } from "rivetkit/client";
import type { Registry } from "./registry";

const client = createClient<Registry>("http://localhost:8080");
```
